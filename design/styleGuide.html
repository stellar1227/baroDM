natural.min.js 변경했습니다. -> ui쪽의 popup객체의 loadContent method에 submit 후 받아온 page String을 div.popup_layout 기준으로 잘라서 jQuery 객체 변환 후 context에 사용합니다. 

    

        

            저는 팝업입지요 'ㅅ' 네

    <div id="popup" style="display:none">
        <div style="width:300px; font-size:18px; padding:20px">
            저는 팝업입지요 'ㅅ' 네 
            alert은 다른놈입니다.
        </div>
    </div>
    <!-- 팝업용 외부 페이지 마크업은 아래 구조를 꼭 지켜주세요 inner_popup부터 포함해서 parsing 됩니다 -->

        <div class="inner_popup">
        <!-- popup contents-->
        </div>
    </div> 
    
    
    <script type="text/javascript">
        $("#popOpen").click(function(){
            
            /*페이지 내부의 블록을 팝업으로 열기*/

                title : '내부팝업',

            }).open();

                    "ko_KR" : {

                    }
                }
            }).open();
        })
    </script>
